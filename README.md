/*Swapping-of-two-numbers*/
#include<stdio.h>
int main()
{
    int x=21,y=27;
    printf("BEFORE SWAPPING\n");
    printf("The value of x is %d\n",x);
    printf("The value of y is %d\n",y);
    printf("AFTER SWAPPING\n");
    printf("The value of x is %d\n",y);
    printf("The value of y is %d\n",x);
}
